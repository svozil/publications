\begin{abstract}
we shall reconsider a system with three observables $A$,$B$ and $C$ such that
\begin{itemize}
\item
$[A,B]=0$, as well as
\item
$[A,C]=0$,
whereas
\item
$[B,C]\neq 0$.
\end{itemize}
\end{abstract}
